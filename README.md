    * pyaudio

        'conda install -c conda-forge pydub'
    * ffmpeg
        'conda install -c conda-forge ffmpeg'
    * numpy
        'conda install -c anaconda numpy'
    * matplotlib
        'conda install -c conda-forge matplotlib'

    * 'conda py37_pyaudio export > py37_pyaudio.yaml'
